<script>
// Hacker Digital Clock
function updateHackerClock() {
  const now = new Date();
  const timeString = now.toLocaleTimeString('en-US', {
    hour12: false,
    hour: '2-digit',
    minute: '2-digit',
    second: '2-digit'
  });

  const dateString = now.toLocaleDateString('en-US', {
    weekday: 'short',
    month: 'short',
    day: '2-digit',
    year: 'numeric'
  }).toUpperCase();

  document.getElementById('hacker-time').textContent = timeString;
  document.getElementById('hacker-date').textContent = dateString;
}

setInterval(updateHackerClock, 1000);
updateHackerClock();

// Uptime counter
let seconds = 413;

function updateUptime() {
  const hours = Math.floor(seconds / 3600);
  const minutes = Math.floor((seconds % 3600) / 60);
  const secs = seconds % 60;

  const timeString =
    String(hours).padStart(2, '0') + ':' +
    String(minutes).padStart(2, '0') + ':' +
    String(secs).padStart(2, '0');

  document.getElementById('uptime').textContent = timeString;
  seconds++;
}

setInterval(updateUptime, 1000);

// Shortcuts toggle
function toggleShortcuts() {
  const shortcuts = document.getElementById('shortcuts');
  const arrow = document.getElementById('arrow');

  shortcuts.classList.toggle('expanded');
  arrow.classList.toggle('collapsed');
}

// Terminal functionality
const commandInput = document.getElementById('command-input');
const commandHistory = document.getElementById('command-history');
let historyIndex = -1;
let commandLog = [];

const commands = {
  help: () => `<div class="help-section">
  <div class="help-command">Available Commands:</div>
  <div class="help-description">help - Show this help message</div>
  <div class="help-description">about - Information about Creed's Domain</div>
  <div class="help-description">whoami - Display current user information</div>
  <div class="help-description">ls - List directory contents</div>
  <div class="help-description">pwd - Print working directory</div>
  <div class="help-description">date - Display current date and time</div>
  <div class="help-description">clear - Clear the terminal screen</div>
  <div class="help-description">matrix - Enter the matrix...</div>
  <div class="help-description">hack - Initiate hacking sequence</div>
  <div class="help-description">status - System status report</div>
  <div class="help-description">contact - How to reach Creed</div>
  <div class="help-description">shortcuts - Toggle access shortcuts panel</div>
</div>`,

  about: () => `<div class="info">Welcome to Creed's Domain...</div>`,

  whoami: () => `<div class="success">creed<br>Domain Administrator</div>`,

  ls: () => `<div class="output">projects/<br>documents/<br>shortcuts/<br>README.txt<br>secrets.encrypted</div>`,

  pwd: () => '<div class="output">/home/creed/domain</div>',

  date: () => `<div class="output">${new Date().toString()}</div>`,

  clear: () => {
    commandHistory.innerHTML = '';
    return '';
  },

  shortcuts: () => {
    toggleShortcuts();
    return '<div class="success">Access shortcuts panel toggled.</div>';
  },

  matrix: () => {
    let output = '<div class="success">Entering the Matrix...</div>';
    let chars = '01';
    for (let i = 0; i < 8; i++) {
      let line = '';
      for (let j = 0; j < 50; j++) {
        line += chars[Math.floor(Math.random() * chars.length)];
      }
      output += `<div style="color: #00ff41;">${line}</div>`;
    }
    output += '<div class="success">Welcome to the Matrix, Neo.</div>';
    return output;
  },

  hack: () => `<div class="warning">Initiating hack sequence...</div>`,

  status: () => `<div class="info">System Status: All systems operational</div>`,

  contact: () => `<div class="success">Contact Creed via encrypted channels</div>`
};

function executeCommand(cmd) {
  const trimmedCmd = cmd.trim().toLowerCase();
  if (commands[trimmedCmd]) {
    return commands[trimmedCmd]();
  } else if (trimmedCmd === '') {
    return '';
  } else {
    return `<div class="error">Command not found: ${cmd}</div>`;
  }
}

function addToHistory(command, output) {
  const historyItem = document.createElement('div');
  historyItem.innerHTML = `
    <div class="command-line">
      <span class="prompt"><span class="user">creed</span>@<span class="domain">domain</span>:~$</span>
      <span>${command}</span>
    </div>
    ${output}
  `;
  commandHistory.appendChild(historyItem);
  commandHistory.scrollTop = commandHistory.scrollHeight;
}

function runTerminalCommand(cmd) {
  const output = executeCommand(cmd);
  addToHistory(cmd, output);
  commandInput.focus();
}

commandInput.addEventListener('keydown', (e) => {
  if (e.key === 'Enter') {
    const command = commandInput.value;
    const output = executeCommand(command);
    addToHistory(command, output);
    commandLog.push(command);
    historyIndex = commandLog.length;
    commandInput.value = '';
  } else if (e.key === 'ArrowUp') {
    e.preventDefault();
    if (historyIndex > 0) {
      historyIndex--;
      commandInput.value = commandLog[historyIndex];
    }
  } else if (e.key === 'ArrowDown') {
    e.preventDefault();
    if (historyIndex < commandLog.length - 1) {
      historyIndex++;
      commandInput.value = commandLog[historyIndex];
    } else {
      historyIndex = commandLog.length;
      commandInput.value = '';
    }
  }
});

commandInput.focus();
document.addEventListener('click', () => commandInput.focus());
</script>
